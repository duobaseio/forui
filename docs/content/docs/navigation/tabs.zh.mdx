import tabsUsage from '@/snippets/usages/widgets/tabs/tabs.json';

### 可滑动

  在桌面端和 web 端，按住 `Shift` 滚动即可在标签页之间滑动切换。
</Callout>

<Tabs items={['预览', '代码']}>

    <Widget name='tabs' variant='swipeable' height={500}/>

  <Tab>
    <CodeSnippet snippet={swipeableSnippet} />
  </Tab>
</Tabs>
